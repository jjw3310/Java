1.테이블 작성

CREATE TABLE BOARD 
(
  BNO NUMBER NOT NULL 
, TITLE VARCHAR2(200 BYTE) NOT NULL 
, CONTENT CLOB 
, WRITER VARCHAR2(30 BYTE) NOT NULL 
, REGDATE DATE DEFAULT sysdate 
, VIEWCNT NUMBER DEFAULT 0 
, SHOW CHAR(1 BYTE) DEFAULT 'Y' 
);
CREATE TABLE MEMBER 
(
  MEMBER_ID VARCHAR2(20 BYTE) NOT NULL 
, MEMBER_NAME VARCHAR2(20 BYTE) NOT NULL 
, MEMBER_PWD VARCHAR2(500 BYTE) NOT NULL 
, MEMBER_YEAR DATE DEFAULT NULL NOT NULL 
, MEMBER_DATE DATE DEFAULT sysdate NOT NULL 
, MEMBER_EMAIL VARCHAR2(40 BYTE) 
, MEMBER_PHONE VARCHAR2(20 BYTE) 
, MEMBER_SEQ NUMBER NOT NULL 
, MEMBER_ZIPCODE VARCHAR2(20 BYTE) NOT NULL 
, MEMBER_FADDR VARCHAR2(350 BYTE) NOT NULL 
, MEMBER_LADDR VARCHAR2(350 BYTE) NOT NULL 
);
CREATE TABLE PRODUCT 
(
  CATE_ID VARCHAR2(20 BYTE) NOT NULL 
, CATE_NAME VARCHAR2(50 BYTE) NOT NULL 
);
CREATE TABLE PRODUCT_CART 
(
  MEMBER_ID VARCHAR2(20 BYTE) NOT NULL 
, PRD_ID VARCHAR2(20 BYTE) NOT NULL 
, BUY_QUANTITY NUMBER DEFAULT 1 
, CART_ID NUMBER NOT NULL 
, PRD_OPT VARCHAR2(50 BYTE) 
);
CREATE TABLE PRODUCT_INFO 
(
  CATE_ID VARCHAR2(20 BYTE) NOT NULL 
, PRD_DATE DATE DEFAULT sysdate 
, PRD_NAME VARCHAR2(50 BYTE) 
, PRD_ID VARCHAR2(20 BYTE) NOT NULL 
, PRD_PRICE NUMBER DEFAULT 0 
, PRD_IMG VARCHAR2(50 BYTE) 
, PRD_DELIVERY NUMBER DEFAULT 0 
, PRD_STOCK NUMBER DEFAULT 0 
, PRD_OPTION VARCHAR2(50 BYTE) 
, PRD_DISCOUNT NUMBER 
);
CREATE TABLE PRODUCT_PAYMENT 
(
  PAYMENT_ID NUMBER NOT NULL 
, MEMBER_ID VARCHAR2(20 BYTE) NOT NULL 
, PRD_ID VARCHAR2(20 BYTE) NOT NULL 
, BUY_QUANTITY NUMBER NOT NULL 
, MEMBER_ZIPCODE VARCHAR2(20 BYTE) NOT NULL 
, MEMBER_FADDR VARCHAR2(350 BYTE) NOT NULL 
, MEMBER_LADDR VARCHAR2(350 BYTE) NOT NULL 
, MEMBER_PHONE VARCHAR2(20 BYTE) 
, CREDIT_NUMBER VARCHAR2(16 BYTE) NOT NULL 
, PAYMENT_DATE DATE DEFAULT SYSDATE NOT NULL 
, DELIVERY VARCHAR2(20 BYTE) 
);
commit;


CREATE SEQUENCE "BOARD_SEQ" INCREMENT BY 1 START WITH 1;
CREATE SEQUENCE "CART_ID_SEQ" INCREMENT BY 1 START WITH 1;
CREATE SEQUENCE "ID_SEQ" INCREMENT BY 1 START WITH 1;
CREATE SEQUENCE "REPLY_SEQ" INCREMENT BY 1 START WITH 1;
